```bash
cd 03-infra-k8s/ingress-nginx/dev
./install.sh
```

### 最小化 Ingress 示例
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  namespace: default
spec:
  ingressClassName: nginx
  rules:
  - host: my-app.czw-sre.internal

      paths:
      - path: /
        pathType: Prefix
        backend:

            name: my-app
            port:
```

## 2. 高级配置项

### 2.1 ConfigMap 配置
修改 `ingress-nginx-controller` ConfigMap：

```yaml
data:
  # 真实客户端 IP 透传（已默认配置）
  use-forwarded-headers: "true"
  compute-full-forwarded-for: "true"
  forwarded-for-header: "X-Forwarded-For"
  real-ip-header: "X-Forwarded-For"
  real-ip-recursive: "true"

  # 日志格式
  log-format-upstream: '$remote_addr - $remote_user [$time_local] "$request" $status $body_bytes_sent "$http_referer" "$http_user_agent" $request_length $request_time [$proxy_upstream_name] [$proxy_alternative_upstream_name] $upstream_addr $upstream_response_length $upstream_response_time $upstream_status $req_id'

  # 日志级别（debug, info, notice, warn, error）
  error-log-level: "notice"

  # 连接超时
  proxy-connect-timeout: "5"
  proxy-send-timeout: "60"
  proxy-read-timeout: "60"

  # 缓冲区大小
  proxy-body-size: "1m"
  client-body-buffer-size: "32k"
  client-header-buffer-size: "1k"
  large-client-header-buffers: "4 8k"

  # Gzip 压缩
  gzip-level: "5"
  gzip-types: "application/atom+xml application/javascript application/x-javascript application/json application/rss+xml application/vnd.ms-fontobject application/x-font-ttf application/x-web-app-manifest+json application/xhtml+xml application/xml font/opentype image/svg+xml image/x-icon text/css text/plain text/x-component"

  # SSL 配置
  ssl-protocols: "TLSv1.2 TLSv1.3"
  ssl-prefer-server-ciphers: "true"

  # 限流
  limit-connections: "100"
  limit-rps: "100"
```

### 2.2 修改 ConfigMap
```bash
# 编辑 ConfigMap
kubectl edit configmap -n ingress-nginx ingress-nginx-controller

# 验证
kubectl get configmap -n ingress-nginx ingress-nginx-controller -o yaml
```

## 3. 常用注解

### 3.1 通用注解
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    # 重定向 HTTP 到 HTTPS
    nginx.ingress.kubernetes.io/ssl-redirect: "true"

    # 强制 SSL
    nginx.ingress.kubernetes.io/force-ssl-redirect: "true"

    # 超时配置
    nginx.ingress.kubernetes.io/proxy-connect-timeout: "5"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "60"
    nginx.ingress.kubernetes.io/proxy-read-timeout: "60"

    # Body 大小限制
    nginx.ingress.kubernetes.io/proxy-body-size: "10m"

    # 缓冲区
    nginx.ingress.kubernetes.io/proxy-buffering: "on"
    nginx.ingress.kubernetes.io/proxy-buffer-size: "4k"
    nginx.ingress.kubernetes.io/proxy-buffers-number: "8"

    # CORS
    nginx.ingress.kubernetes.io/enable-cors: "true"
    nginx.ingress.kubernetes.io/cors-allow-origin: "*"
    nginx.ingress.kubernetes.io/cors-allow-methods: "GET, POST, OPTIONS"
    nginx.ingress.kubernetes.io/cors-allow-headers: "DNT,X-CustomHeader,Keep-Alive,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Authorization"

    # 重写规则
    nginx.ingress.kubernetes.io/rewrite-target: "/$1"
    nginx.ingress.kubernetes.io/app-root: "/app"

    # 服务别名
    nginx.ingress.kubernetes.io/server-alias: "alias.czw-sre.internal"
```

### 3.2 限流注解
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    # 每秒请求数限制（按 IP）
    nginx.ingress.kubernetes.io/limit-rps: "10"

    # 每分钟请求数限制（按 IP）
    nginx.ingress.kubernetes.io/limit-rpm: "600"

    # 连接数限制（按 IP）
    nginx.ingress.kubernetes.io/limit-connections: "10"

    # 限流区域名称
    nginx.ingress.kubernetes.io/limit-rate: "100k"
```

### 3.3 认证注解
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    # Basic Auth（需要先创建 Secret）
    nginx.ingress.kubernetes.io/auth-type: basic
    nginx.ingress.kubernetes.io/auth-secret: basic-auth
    nginx.ingress.kubernetes.io/auth-realm: "Authentication Required"

    # 外部认证服务
    nginx.ingress.kubernetes.io/auth-url: "https://auth.czw-sre.internal/verify"
    nginx.ingress.kubernetes.io/auth-signin: "https://auth.czw-sre.internal/login?rd=$escaped_request_uri"

    # Auth-TLS
    nginx.ingress.kubernetes.io/auth-tls-secret: "default/auth-tls"
```

### 3.4 负载均衡注解
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    # 负载均衡算法（round_robin, least_conn, ip_hash）
    nginx.ingress.kubernetes.io/load-balance: "least_conn"

    # 会话亲和性（基于 cookie）
    nginx.ingress.kubernetes.io/affinity: "cookie"
    nginx.ingress.kubernetes.io/session-cookie-name: "INGRESSCOOKIE"
    nginx.ingress.kubernetes.io/session-cookie-hash: "sha1"
    nginx.ingress.kubernetes.io/session-cookie-expires: "172800"

    # 后端协议
    nginx.ingress.kubernetes.io/backend-protocol: "HTTPS"
```

## 4. TLS 配置

### 4.1 自动 TLS（cert-manager）
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    cert-manager.io/cluster-issuer: "letsencrypt-prod"
spec:
  ingressClassName: nginx
  tls:
  - hosts:
    - my-app.czw-sre.internal
    secretName: my-app-tls
  rules:
  - host: my-app.czw-sre.internal
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: my-app
            port:
              number: 80
```

### 4.2 手动 TLS
```bash
# 创建证书
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout tls.key -out tls.crt -subj "/CN=my-app.czw-sre.internal/O=my-app"

# 创建 Secret
kubectl create secret tls my-app-tls --key tls.key --cert tls.crt
```

```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
spec:
  ingressClassName: nginx
  tls:
  - hosts:
    - my-app.czw-sre.internal
    secretName: my-app-tls
  rules:
  - host: my-app.czw-sre.internal
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: my-app
            port:
              number: 80
```

### 4.3 TLS 增强配置
```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-app
  annotations:
    # HSTS
    nginx.ingress.kubernetes.io/ssl-redirect: "true"
    nginx.ingress.kubernetes.io/force-ssl-redirect: "true"
    nginx.ingress.kubernetes.io/hsts: "true"
    nginx.ingress.kubernetes.io/hsts-include-subdomains: "true"
    nginx.ingress.kubernetes.io/hsts-max-age: "31536000"
    nginx.ingress.kubernetes.io/hsts-preload: "true"
```

## 7. 日志配置

### 7.1 查看日志
```bash
# 查看 controller 日志
kubectl logs -n ingress-nginx -l app.kubernetes.io/name=ingress-nginx -f

# 查看特定 Pod 日志
kubectl logs -n ingress-nginx <pod-name> -f
```

### 7.2 日志格式
默认的 `log-format-upstream` 包含：
- `$remote_addr`：客户端 IP
- `$time_local`：本地时间
- `$request`：请求行
- `$status`：响应状态码
- `$body_bytes_sent`：发送的字节数
- `$http_referer`：Referer
- `$http_user_agent`：User Agent
- `$request_time`：请求处理时间
- `$upstream_addr`：后端地址
- `$upstream_response_time`：后端响应时间
- `$upstream_status`：后端状态码

### 7.3 自定义日志格式
```yaml
data:
  log-format-upstream: '$remote_addr - $remote_user [$time_local] "$request" $status $body_bytes_sent "$http_referer" "$http_user_agent" $request_length $request_time [$proxy_upstream_name] [$proxy_alternative_upstream_name] $upstream_addr $upstream_response_length $upstream_response_time $upstream_status $req_id'
```

## 8. 监控与指标

### 8.1 Prometheus 指标
```bash
# 查看指标
kubectl port-forward -n ingress-nginx svc/ingress-nginx-controller 10254:10254
curl http://localhost:10254/metrics
```

### 8.2 关键指标
- `nginx_ingress_controller_requests`：请求总数
- `nginx_ingress_controller_request_duration_seconds`：请求耗时
- `nginx_ingress_controller_connections`：连接数
- `nginx_ingress_controller_success`：成功请求数
- `nginx_ingress_controller_ingress_upstream_latency_seconds`：上游延迟
