## Pod / 实例状态

```bash
# 所有 Redis Pod
kubectl get pods -n redis -o wide

# Standalone
kubectl get pods -n redis -l app=redis-standalone

# Sentinel-HA (Redis + Sentinel)
kubectl get pods -n redis -l app=redis-replication
kubectl get pods -n redis -l redis_setup_type=sentinel

# Cluster (leader + follower)
kubectl get pods -n redis -l redis_cluster=redis-cluster

# CR 状态
kubectl get redis -A
kubectl get redisreplication -A
kubectl get rediscluster -A
```

## Key 操作

```bash
# 基础
SET key value
SET key value EX 3600
GET key
DEL key1 key2
EXISTS key
TYPE key

# 过期
EXPIRE key 60
TTL key
PERSIST key

# 查找（生产用 SCAN 代替 KEYS）
KEYS prefix:*
SCAN 0 COUNT 100 MATCH prefix:*

# 统计
DBSIZE
RANDOMKEY
```

## 日常运维操作

```bash
# 修改密码（生成长密码）
PASSWORD=$(openssl rand -base64 16)
for mode in standalone sentinel-ha cluster; do
  sed -i '' "s/password:.*/password: $PASSWORD/" operator/$mode/00-secret.yaml
done
kubectl apply -f operator/standalone/00-secret.yaml
kubectl apply -f operator/sentinel-ha/00-secret.yaml
kubectl apply -f operator/cluster/00-secret.yaml
kubectl rollout restart sts -n redis --all

# 重启 Pod（配置变更后）
kubectl rollout restart sts -n redis redis-standalone
kubectl rollout restart sts -n redis redis-replication
kubectl rollout restart sts -n redis redis-cluster-leader redis-cluster-follower

# 查看日志
kubectl logs -n redis <pod> redis
kubectl logs -n redis <pod> redis-exporter

# 进入 Pod 的 Redis-CLI
kubectl exec -it -n redis <pod> -- redis-cli -a 'redis@czw123'

# 读取 Pod 中 redis.conf
kubectl exec -n redis <pod> -- cat /etc/redis/redis.conf

# 查看 AOF/RDB 文件大小
kubectl exec -n redis <pod> -- du -sh /data/*.aof /data/*.rdb

# 触发 AOF 重写
redis-cli -p 30003 -a 'redis@czw123' BGREWRITEAOF

# 查看 operator 日志
kubectl logs -n redis-operator deploy/redis-operator | tail -50
```
